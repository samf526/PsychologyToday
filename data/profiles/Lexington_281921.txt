$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});

                        <h3>Your Recovery Starts Here</h3>
                <div class="statementPara">
            Our mission at Palm Beach Recovery and Wellness is to see every client through their own personal recovery process and into a healthy sober lifestyle. Everyone that walks through our doors is given a unique level of care dependent on their specific needs. Our hand-picked certified therapists and counselors have a passion for helping those struggling with addiction and are adept at handling all forms of substance abuse. We truly believe that no case is hopeless and that anyone, no matter the circumstances, deserves healing and a fair shot at sobriety! It is known that substance abuse affects the user on all fronts and that long-term recovery is typically found when the mind, spirit and body are addressed simultaneously. We have found that focusing strictly on any of these categories is but a temporary solution. At Palm Beach Recovery and Wellness, we treat more than just the symptoms of the disease. Get the results you need through cutting edge therapeutic techniques designed to get to the root of the problem and start the healing immediately.
        </div>
    
                        <h3>Addiction Treatment: A Continuum of Care</h3>
                <div class="statementPara">
            Here at Palm Beach Recovery and Wellness, we specialize in outpatient addiction treatment and sober living. Most of our clients come to us after being referred to another treatment facility for medical detox and inpatient residential treatment. After the traditional 30 day inpatient treatment our clients come to us for outpatient aftercare and certified sober living. Our Intensive Outpatient Program consists of three sessions per week. During these clinical sessions our clients are given a one-on-one session with their therapist, group sessions, and long term psychiatric and medical care. While attending our outpatient facility, clients are encouraged to live in our certified sober living homes. These homes are managed 24/7 and have are designed to give our recovering clients accountability and structure.
        </div>
    
                        <h3>Our Commitment to Quality Care</h3>
                <div class="statementPara">
            Our staff of clinicians, doctors, and behavioral health technicians value one thing: to help those suffering from addiction get the help and love that they deserve. All of our employees are committed to helping you find your path to long term sobriety.
