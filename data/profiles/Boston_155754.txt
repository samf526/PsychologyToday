$(document).ready(function(){

            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',

            });
            $('#subBrandsXsm').removeClass('show');

            $('#subBrandXsm').removeClass('closing');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});

                    <div class="statementPara">
            My private practice is dedicated to partnering with individuals (primarily adults) who are on a growth path and seeking a deeper sense of clarity, purpose, and inspiration in their lives.  As I have trained extensively in various treatment modalities, I utilize a heart-centered, humanistic and holistic approach to promote insight, ease and empowerment.  My 15+ years of clinical experience includes extensive work with children, adolescents, adults, families and older adults who have suffered from depression, anxiety, addiction, and trauma.  I also have significant experience working with those struggling with substance abuse, infertility and relationship issues.
        </div>
    
                    <div class="statementPara">
            My training includes eye movement desensitization and reprocessing (EMDR), cognitive behavioral therapy (CBT) and mindfulness-based stress reduction (MBSR).  In addition, I have trained in enneagram studies of the narrative tradition (ESNT); a personality system that provides a foundational framework for understanding self, relationships and practices for interpersonal awareness, growth and development.
        </div>
