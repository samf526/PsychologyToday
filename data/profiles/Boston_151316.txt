$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});

                        <h3>Effective Youth Mental Health Services</h3>
                <div class="statementPara">
            The Center for Effective Child Therapy at Judge Baker provides mental health assessments and focused short-term treatments for children and their families. We use only &quot;evidence-based treatments,&quot; such as cognitive behavioral therapy (CBT), that have been scientifically proven to work. At CECT, we continually monitor your child&#039;s progress in order to individualize treatment and maximize gains. We do this by both using proven assessment tools and listening closely to the children and families we work with, ensuring that everyone takes an active part in goal-setting and treatment. These careful assessments help us go far beyond a child&#039;s diagnosis; we can focus on what the child and family hope to change. One of our primary goals is to help children develop resiliency - an internal sense of stability that allows them to face the challenges of daily life. It&#039;s the sense of empowerment children have when they realize they have control over their feelings and behavior, and enhances their ability to grow and thrive.
        </div>
    
                        <h3>Our Toolbox</h3>
                <div class="statementPara">
            At CECT, we equip children and their parents with a toolbox of skills to successfully navigate the inevitable challenges and stresses life presents. Our goal is to teach a range of techniques to develop healthier behaviors and thoughts, overcome obstacles, build confidence, and develop resiliency. We aim to give children and families the skills to meet their current challenges and to help them thrive throughout all the stages of development.
        </div>
    
                        <h3>A History of Excellence</h3>
                <div class="statementPara">
            Founded in 1917, Judge Baker Children&#039;s Center is a Harvard Medical School affiliate with proven leadership in children&#039;s mental health. Committed to bridging the current gap between research and practice, Judge Baker integrates research, professional training and direct service. In addition to its research portfolio, JBCC operates programs in special education, outpatient mental health, community and school consultation, specialized summer treatment, and child welfare practice - all with the goal of helping children and families chart their own best course to grow and thrive. The Center for Effective Child Therapy focuses on care that works for children using evidence-based practices such as cognitive behavioral therapy (CBT). This means that we help children and families using effective, sustainable treatments that have withstood rigorous clinical testing. We measure and track results to hold ourselves accountable to children and families. Our focus on data and quantifiable measures gives us unique insight into the best ways to help children and families find - and stay on - the best course for them.

        function contactBar() {
            var windowWidth = $(window).width();
            if (windowWidth > 768) {
               var height = $('#profHdr').height() + 115;
                if (window.pageYOffset > height) {
                        showContactBar();
                    } else if (window.pageYOffset <= height) {
                        hideContactBar();
                    }
            } else {
                var height = $('#profHdr').height() + 40;
                if (window.pageYOffset > height) {
                    showContactBar();
                } else if (window.pageYOffset <= height) {
                    hideContactBar();
                }
            }
        }
        function setContactBarPosition() {
            var offset = $('#contactBar').outerHeight() - $('#headerShadow').outerHeight();
            $('#contactBar').css({
                "top": '-' + offset + 'px'
            });
        }
        function showContactBar() {
            var offset = $('#contactBar').outerHeight();
            $('#headerShadow').css({
                "top": offset + 'px'
            });
        }
        function hideContactBar() {
            $('#headerShadow').css({
                "top": '0px'
            });
        }
