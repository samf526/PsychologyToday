$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});

    function initializeMap(e) {
        var count = 0;
        var markers = [];
        var markerABC = 'A';
        var coords;
        var geocoder = new google.maps.Geocoder();
        var latlng = new google.maps.LatLng(44.8819, -93.2217);

            mapTypeId: google.maps.MapTypeId.ROADMAP

        $(addrs).each(function () {

            over_enable = parseInt($(this).attr('data-map-over-enable'));
            address = $(this).attr('data-' + $(this).attr('data-map-address'));
            var markerABC = 'A';

            var addrHtml = $(this).parents('.address').clone().appendTo('#map-addresses');
            $('.address-buttons', addrHtml).remove();

            markerABC = String.fromCharCode(markerABC.charCodeAt() + count);

            if (over_enable) {
                coords = new google.maps.LatLng(lat, lon);
                map.setCenter(coords);
                marker = new google.maps.Marker({
                    icon: '//www.google.com/mapfiles/marker' + markerABC + '.png',
                    position: coords,
                    animation: google.maps.Animation.DROP,
                    map: map
                });

                markers.push(marker);
                setMap(markers, map);
            } else {
                codeAddress(address, function (coords) {
                    map.setCenter(coords);
                    marker = new google.maps.Marker({
                        icon: '//www.google.com/mapfiles/marker' + markerABC + '.png',
                        position: coords,
                        animation: google.maps.Animation.DROP,
                        map: map
                    });

                    markers.push(marker);
                    setMap(markers, map);
                });
            }
            count++;
        });
    }

    function codeAddress(address, callback) {
        var geocoder = new google.maps.Geocoder();
        if (geocoder) {
            geocoder.geocode({'address': address}, function (results, status) {
                if (status == google.maps.GeocoderStatus.OK) {
                    callback(results[0].geometry.location);
                }
            });
        }
    }

    function setMap(markers, map) {
        zoomChangeBoundsListener = google.maps.event.addListener(map, 'bounds_changed', function (event) {
