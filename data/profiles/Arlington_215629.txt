$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});

                        <h3>Our Program Model</h3>
                <div class="statementPara">
            Ellenhorn is the most comprehensive and integrative private community-based mental health program in the country. Geared to individuals 18 years and older, our approach combines robust psychiatric services with intensive psychosocial care, focusing equally on reducing the psychosocial damage caused by stigma, loss of social identity and institutionalization, as on treating psychiatric illness and co-occurring disorders. Using the best-practice &quot;gold standard&quot; community-based treatment model, Program for Assertive Community Treatment (PACT), our multidisciplinary teams of professionals - which include psychiatrists and a broad spectrum of other experts - provide rehabilitation and treatment resources to clients living, for the most part, in apartments in the community. A small number of clients receive services at our staff-supervised home called &quot;The Brackett House&quot;.
        </div>
    
                        <h3>Our Beliefs</h3>
                <div class="statementPara">
            Our clients come with a wide range of psychiatric issues, but all have a common and serious complaint: the loss of their dreams. Their lives interrupted by their illness, they feel as if life itself has failed them, or that they have failed at life. Accordingly, the recovery of a sense of hope along with forward momentum in one&#039;s life, are central parts of our work with each client. Our job is to provide top-quality clinical care, with a focus on the psychosocial goals of regaining a sense of mastery and a return to the pursuit of life dreams.   We seek to re-empower our clients, helping them regain a sense of voice and control in their lives. This empowerment begins immediately, through their relationship with us. As our partner, each client helps set the course in his or her recovery.  We welcome the chance to talk with you about whether Ellenhorn can help. (800) 515-9972. Or email us at: info@ellenhorn.com
        </div>
    
                        <h3>Rehabilitation &amp; Recovery</h3>
                <div class="statementPara">
            As a psychiatric and psychosocial program, we are as focused on ameliorating the ravages of stigma, loss of social role and institutionalization, as we are focused on treating psychiatric illnesses. Autonomy and mastery are coupled with symptom reduction as primary goals for clients. Focused on individual empowerment, we treat our clients as autonomous agents in their lives, and thus deserving of an individualized outlook. We empower our clients by appreciating their unique struggles and strengths. This individualized approach is the very counter-definition of institutional care. We do not provide care in bulk. We contour our care to the individual needs of our clients.  We often say at Ellenhorn that our clients come with a wide-range of psychiatric issues, but they all suffer from one serious complaint: losing their dreams. Their lives interrupted by their illness, they feel like life itself has failed them, or that they, themselves have failed at life. Accordingly, hope, the struggle over hope, and the means to recover a sense of hope are central parts of our work. It is our job to dream of possibilities with our clients, to seek out and find these opportunities, and, in the end, to help our clients achieve their greatest potential.
        </div>

        function contactBar() {
            var windowWidth = $(window).width();
            if (windowWidth > 768) {
               var height = $('#profHdr').height() + 115;
                if (window.pageYOffset > height) {
                        showContactBar();
                    } else if (window.pageYOffset <= height) {
                        hideContactBar();
                    }
            } else {
                var height = $('#profHdr').height() + 40;
                if (window.pageYOffset > height) {
                    showContactBar();
                } else if (window.pageYOffset <= height) {
                    hideContactBar();
                }
            }
        }
        function setContactBarPosition() {
            var offset = $('#contactBar').outerHeight() - $('#headerShadow').outerHeight();
            $('#contactBar').css({
                "top": '-' + offset + 'px'
            });
        }

            var offset = $('#contactBar').outerHeight();
            $('#headerShadow').css({
                "top": offset + 'px'
            });
        }
        function hideContactBar() {
            $('#headerShadow').css({
                "top": '0px'
            });
