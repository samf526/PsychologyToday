$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});
</script>

                        <h3>Cognitive Behavioral Therapy Specialists</h3>
                <div class="statementPara">
            Life Changes Group is an integrative therapy, psychopharmacology, and consulting practice dedicated to providing the highest quality of psychological care for children, adolescents, adults, and seniors. Our friendly and experienced team of doctors offers the best, most evidence-based practices in the treatment of psychiatric illness, neurocognitive assessment, educational testing, and school consultation. Our clinicians focus on the integration of treatment, education, collaboration, and partnership to help individuals, couples, and families reach their goals and improve the quality of their lives. Our main focus is Cognitive Behavioral Therapy, and we are one of the few practices in the Boston/Cambridge area to offer specific, guided, and collaborative CBT support for clients of all ages and backgrounds. If you have suffered from a anxiety or depression, and have found the typical psychotherapy approach ineffective, our clinicians offer specific strategies to reduce your symptoms. We pride ourselves on using collaborative, research-supported models for returning clients to their normal levels of functioning, including cognitive reframing, stress-reduction (e.g., diaphragmatic breathing, meditation, progressive muscle relaxation), guided exposure, and biofeedback for understanding your body&#039;s response to stress. We combine compassionate listening and active intervention to produce lasting, life-changing results.
        </div>
    
                        <h3>Our Philosophy</h3>
                <div class="statementPara">
            At Life Changes Group we recognize that finding appropriate behavioral healthcare can be a daunting task. Many of you have already spent hours or days searching online, calling insurance companies, meeting with general practitioners, or asking friends if they &quot;know of anyone&quot; who can help you make the changes you desire. Seeking behavioral healthcare is challenging enough without these additional hurdles. Our goal at Life Changes Group is to simplify the process of looking for appropriate treatment and accelerate the process of getting appropriate treatment. We appreciate that many of our clients are at a crossroads in their lives, already feeling vulnerable, confused, and unsure whether change is possible. The shared philosophy of our clinicians is that change is possible, and through education, collaboration, and thoughtful treatment goals everyone is capable of finding that change. We develop an individualized and structured recovery plan for each client, focused on the biological, social, and emotional foundations of the specific problem, including: 1) Comprehensive intake and assessment 2) Collaborative goal-setting and measurable objectives 3) Psychoeducation and literature review 4) Symptom-focused treatment and weekly monitoring 5) Collaboration with other health professionals 6) Family education and support 7) Relapse prevention
