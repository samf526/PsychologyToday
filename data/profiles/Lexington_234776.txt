$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});
</script>

                        <h3>What we do</h3>
                <div class="statementPara">
            We treat addictions and all problems warranting change with the most cutting edge, evidence-based therapies and medications, delivered by kind, approachable people. We specialize in complex, combined addictions and psychiatric issues including bipolar and other mood disorders, opiate, alcohol and other addictions, personality disorders and more. A team that might include a psychiatrist, nurse, therapist, family therapist and case manager help you figure out the best way to improve safety and functioning, and set the course for lasting change. We try to keep you moving forwards, limiting inpatient detox or hospitalization through support and sophisticated pharmacology. Because force creates resistance, we enhance your motivation, determination and skill through reason, compassion, and step-wise success. We work on many levels, depending on where your strengths and preferences are and what feels the greatest bang for the buck: neurobiological, behavioral, emotional, cognitive, vocational, social and existential. We work efficiently, but know that sometimes rapid transformation is achieved when we slow down, attend, feel. We want you to leave feeling better, being better at feeling, and with biology, skills and relationships that will sustain your recovery.
        </div>
    
                        <h3>How We&#039;re Different</h3>
                <div class="statementPara">
            We are scientific rather than dogmatic. We embrace harm reduction and moderation training. Our interventions are respectful and gentle, not confrontational. The pharmacology is collaborative - you&#039;ll understand why each option is presented and we&#039;ll work with you whatever path you choose. The therapy is tailored not one fits all. Sometimes we use DBT or contingency management, sometimes AEDP (experiential and emotion focused), sometimes in-the-field coaching, sometimes meditation. Each choice is a product of your preference and our suggestion and is delivered by an expert. For opiate addiction, we strongly advocate medications including buprenorphine and naltrexone over abstinence because decades of research show many lives are saved. But again, we&#039;ll stick with you as you learn what&#039;s right for you right now.
        </div>
    
                        <h3>Our Training</h3>
                <div class="statementPara">
            We only employ deeply trained, deeply experienced, deeply kind people; respected experts in their fields who want to work here because they know the sum is greater than its parts, that complex issues demand complex approaches, and have seen great results from excellent collaborative care. We&#039;ve decades of training (Cornell, Harvard) between us and are all constantly learning new approaches and monitoring our own standards through training, supervision and data. Our staff continue to teach at the university level (Harvard) and nationally.
        </div>

<h2>Psych Garden's Groups</h2>
    <div class="group-small">
        <h2>
            <a href="https://therapists.psychologytoday.com/rms/prof_detail.php?profid=234776&gid=79547&sid=1487100109.9878_29182&city=Lexington&state=MA&rec_next=1">DBT for addictions</a>
        </h2>

        <div class="group-description">
            <p>This group is for people, whose emotions get the better of them, disrupting plans, relationships, thinking and contributing to impulsivity including drug or alcohol use. Each session we will cover a set of skills aimed at managing urges and cravings, avoiding opportunities and cues to use, reducing behaviors conducive to drug use, increasing healthy social networks, and coping ahead for stressful events. We&#039;ll work on identifying personal red flags, describing emotional experiences, and helping each other recognize when we are in predictable mind states that lend themselves to use.</p>

        function contactBar() {
            var windowWidth = $(window).width();
            if (windowWidth > 768) {
               var height = $('#profHdr').height() + 115;
                if (window.pageYOffset > height) {
                        showContactBar();
                    } else if (window.pageYOffset <= height) {
                        hideContactBar();
                    }
            } else {
                var height = $('#profHdr').height() + 40;
                if (window.pageYOffset > height) {
                    showContactBar();
                } else if (window.pageYOffset <= height) {
                    hideContactBar();
                }
            }
        }
        function setContactBarPosition() {
            var offset = $('#contactBar').outerHeight() - $('#headerShadow').outerHeight();
            $('#contactBar').css({
                "top": '-' + offset + 'px'
            });
        }

            var offset = $('#contactBar').outerHeight();
            $('#headerShadow').css({
                "top": offset + 'px'
            });
        }
        function hideContactBar() {
            $('#headerShadow').css({
                "top": '0px'
            });
        }
