$(document).ready(function(){

    $('#pageContainer').on(sx.transitionEndEvent, function(e) {

        if (e.target.id != 'pageContainer') {
            return;
        }

        if ($('#subBrandsXsm').hasClass('closing')) {
            $('#headerWrapper').css({
                'position' : 'fixed',
                'top' : '0px'
            });
            $('#subBrandsXsm').removeClass('show');
            $('#subBrandsXsm').addClass('hide');
            $('#subBrandXsm').removeClass('closing');
            $('#pageContainer').removeClass('overlay');

            var scrollPosition = Math.abs(parseInt($('body').css('top'),10));
            $('body').css({
                position: 'static',
                overflow: 'auto'
            });
            $('body').scrollTop(scrollPosition);
        } else {
            $('#subBrandXsm').removeClass('opening');
        } 
    });

    $('#subBrandsBtnXs').on('click', function(e) {
        e.preventDefault();
        if ($('html').hasClass('subBrandsOpenXsm')) {
            $('#subBrandsXsm').removeClass('opening');
            $('#subBrandsXsm').addClass('closing');
            $('html').removeClass('subBrandsOpenXsm');
            $('#pageContainer').off('click.menuDismissal');
        } else {
            $('#pageContainer').on('click.menuDismissal', function(e) {
                e.preventDefault();
                if ($('html').hasClass('subBrandsOpenXsm') ) {
                    if ($('#pageContainer').hasClass('overlay')) {
                        $('#subBrandsXsm').removeClass('opening');
                        $('#subBrandsXsm').addClass('closing');
                        $('html').removeClass('subBrandsOpenXsm');
                        $('#pageContainer').removeClass('overlay');
                        $('#pageContainer').off('click.menuDismissal');
                        $('#pageContainer *').css({ 'pointer-events' : 'auto' });
                    } else {
                       $('#pageContainer').addClass('overlay');
                    }
                } 
            });
            $('#pageContainer *').css({ 'pointer-events' : 'none' });
            $('#subBrandsXsm').addClass('show');
            $('#subBrandsXsm').addClass('opening');
            $('#subBrandsXsm').removeClass('closing');
            $('#headerWrapper').css({
                'position' : 'absolute',
                'top' : $(window).scrollTop() + 'px'
            });
            $('body').css({
                position: 'fixed',
                overflow: 'hidden',
                width: $('body').width(),
                top: '-' + $('body').scrollTop() + 'px'
            })
            $('html').addClass('subBrandsOpenXsm');
        }
    });     
});

                        <h3>New England Recovery &amp; Wellness</h3>
                <div class="statementPara">
            Healing the mind, body, and spirit, New England Recovery &amp; Wellness Center is the premier addiction rehab center located in the heart of New England.
        </div>
    
                        <h3>Clinical Excellence</h3>
                <div class="statementPara">
            Created by a team of experienced addictions professionals, our program offers a curriculum rooted in evidence-based practices. With an interdisciplinary approach that promotes long term sobriety and healing, our team has developed actionable strategies that can be utilized through-out their road to recovery. We believe that sustainability is possible when an individual can successfully shift their internal perceptions and beliefs to a new practice consistent with healthy behaviors and peer support. Our commitment is to helping individuals learn how to live an alcohol and drug free life while creating new possibilities for a positive future. We have a comprehensive approach in treating addiction and co-occurring disorders. We provide a safe, private and therapeutic location for clients to begin or further their recovery. To assure that individuals are treated with dignity and respect through a supportive environment, the individual will be given the tools to build a foundation through educations groups, life skills education, individual therapy and a guided introduction to the 12-steps. Because every addiction and experience differs, we collaborate with the client to develop customized treatment plans based on the assessment made by our clinical team.
        </div>
    
                        <h3>About Us</h3>
                <div class="statementPara">
            At New England Recovery and Wellness, our team is dedicated to providing every client with a recovery experience tailored to their individual needs. The RAW staff understands the road to recovery and the challenges it presents to an individual throughout the journey. Our intimate understanding of the process allows us to give invaluable guidance to all clients, helping them maintain abstinence for not only the immediate future, but for the rest of their lives. Our treatment program is a comprehensive take on recovery, incorporating one on one therapy, group counseling, educational groups on drug and alcohol addiction, and a guided exposure to 12 step programs. Recovery goes further than what we do at our facility. RAW encourages a healthy and happy lifestyle, with nutrition focused catered lunches, daily fitness sessions, yoga, meditation, and a multitude of activities every week. On weekends, enjoy trips to some of the most beautiful sights New England has to offer. What better way to relax and clear your mind than the picturesque hills of New Hampshire?
